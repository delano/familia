    def update h={}
      raise ArgumentError, "Argument to bulk_set must be a hash" unless Hash === h
      data = h.inject([]){ |ret,pair| ret << [pair[0], to_redis(pair[1])] }.flatten
      ret = redis.hmset(rediskey, *data)
      update_expiration
      ret
    end
