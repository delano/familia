    def trace label, redis_client, ident, context=nil
      return unless Familia.debug?
      info "%s (%d:%s): %s" % [label, Thread.current.object_id, redis_client.object_id, ident] 
      info "  +-> %s" % [context].flatten[0..3].join("\n      ") if context
    end
